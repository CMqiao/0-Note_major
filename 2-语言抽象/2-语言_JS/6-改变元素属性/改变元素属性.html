改变元素是否可见： function changVisitAble(id,able){ var node = document.getElementById(id); node.style.display= able; } 改变属性所属类： function changClass(id,classname){ var node = document.getElementById(id); node.setAttribute("class",classname); } function changClassName(id,classname){ var node = document.getElementById(id); node.setAttribute("className",classname); }
